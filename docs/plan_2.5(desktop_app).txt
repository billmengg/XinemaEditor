# Plan 2.5: Desktop App Conversion

## 📋 **Overview**
Convert the Xinema video editor from a web application to a native desktop application. This will enable direct file system access, improved performance, native OS integration, and offline functionality. The conversion will use Electron or Tauri framework to wrap the existing React frontend and Node.js backend into a standalone desktop application.

---

## 🎯 **FRAMEWORK SELECTION**

### **Option 1: Electron (Recommended for MVP)**
**Pros:**
- ✅ Mature ecosystem with extensive documentation
- ✅ Large community and resources
- ✅ Easy integration with existing Node.js backend
- ✅ Full access to Node.js APIs
- ✅ Cross-platform (Windows, macOS, Linux)
- ✅ Extensive plugin ecosystem
- ✅ Easy debugging and development

**Cons:**
- ❌ Larger bundle size (~100-150MB)
- ❌ Higher memory usage
- ❌ Slower startup time compared to native apps

**Best For:** Quick MVP conversion, maximum compatibility, existing Node.js backend

### **Option 2: Tauri (Future Consideration)**
**Pros:**
- ✅ Smaller bundle size (~5-10MB)
- ✅ Lower memory usage
- ✅ Faster performance (native Rust core)
- ✅ Better security model
- ✅ Modern architecture

**Cons:**
- ❌ Requires Rust knowledge for backend
- ❌ Smaller ecosystem
- ❌ More complex setup for Node.js backend
- ❌ Less mature tooling

**Best For:** Production-ready desktop app with performance focus, smaller bundle size

**Recommendation:** Start with **Electron** for MVP, consider **Tauri** migration for v1.0+ if bundle size/performance becomes critical.

#### **1.2 Create Electron Main Process**
Create `electron/main.js`:
```javascript
const { app, BrowserWindow, ipcMain, dialog } = require('electron');
const path = require('path');
const { spawn } = require('child_process');
const fs = require('fs').promises;

let mainWindow;
let backendProcess;

// Start backend server
function startBackend() {
  const backendPath = path.join(__dirname, '../Xinema/backend/server.js');
  backendProcess = spawn('node', [backendPath], {
    cwd: path.join(__dirname, '../Xinema/backend'),
    stdio: 'inherit'
  });
}

// Stop backend server
function stopBackend() {
  if (backendProcess) {
    backendProcess.kill();
  }
}

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1920,
    height: 1080,
    webPreferences: {
      nodeIntegration: false, // Security: don't expose Node.js to renderer
      contextIsolation: true,  // Security: isolate context
      preload: path.join(__dirname, 'preload.js') // Preload script for safe IPC
    },
    icon: path.join(__dirname, '../assets/icon.png') // Optional: app icon
  });

  // Load React app
  const isDev = process.env.NODE_ENV === 'development';
  if (isDev) {
    mainWindow.loadURL('http://localhost:3000');
    mainWindow.webContents.openDevTools();
  } else {
    mainWindow.loadFile(path.join(__dirname, '../Xinema/frontend/build/index.html'));
  }

  mainWindow.on('closed', () => {
    mainWindow = null;
    stopBackend();
  });
}

app.whenReady().then(() => {
  startBackend();
  // Wait for backend to be ready (adjust port if needed)
  setTimeout(() => {
    createWindow();
  }, 2000);

  app.on('activate', () => {
    if (BrowserWindow.getAllWindows().length === 0) {
      createWindow();
    }
  });
});

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    stopBackend();
    app.quit();
  }
});

app.on('before-quit', () => {
  stopBackend();
});

// IPC Handlers for file system access
ipcMain.handle('read-file', async (event, filePath) => {
  try {
    const data = await fs.readFile(filePath);
    return { success: true, data: data.buffer };
  } catch (error) {
    return { success: false, error: error.message };
  }
});

ipcMain.handle('show-open-dialog', async (event, options) => {
  const result = await dialog.showOpenDialog(mainWindow, options);
  return result;
});

ipcMain.handle('show-save-dialog', async (event, options) => {
  const result = await dialog.showSaveDialog(mainWindow, options);
  return result;
});

ipcMain.handle('write-file', async (event, filePath, data) => {
  try {
    await fs.writeFile(filePath, data);
    return { success: true };
  } catch (error) {
    return { success: false, error: error.message };
  }
});
```

#### **1.3 Create Preload Script**
Create `electron/preload.js`:
```javascript
const { contextBridge, ipcRenderer } = require('electron');

// Expose safe APIs to renderer process
contextBridge.exposeInMainWorld('electronAPI', {
  // File system access
  readFile: (filePath) => ipcRenderer.invoke('read-file', filePath),
  writeFile: (filePath, data) => ipcRenderer.invoke('write-file', filePath, data),
  
  // File dialogs
  showOpenDialog: (options) => ipcRenderer.invoke('show-open-dialog', options),
  showSaveDialog: (options) => ipcRenderer.invoke('show-save-dialog', options),
  
  // Platform info
  platform: process.platform,
  version: process.versions.electron
});
```

#### **2.2 Update Imported Media Restoration**
The `restoreImportedMediaFilesDesktop()` function in `App.js` already checks for `window.electronAPI.readFile`, but needs to be updated:

```javascript
const restoreImportedMediaFilesDesktop = async (mediaList) => {
  if (!window.electronAPI || !window.electronAPI.readFile) {
    console.warn('⚠️ Electron API not available');
    return;
  }
  
  try {
    const restoredMedia = await Promise.all(mediaList.map(async (media) => {
      try {
        const filePath = media.path;
        if (!filePath) {
          console.warn(`⚠️ No path for file: ${media.filename}`);
          return { ...media, file: null, url: null, restored: true };
        }
        
        // Use Electron API to read file
        const result = await window.electronAPI.readFile(filePath);
        if (!result.success) {
          throw new Error(result.error);
        }
        
        // Convert ArrayBuffer to File object
        const blob = new Blob([result.data]);
        const file = new File([blob], media.filename, {
          type: 'video/mp4',
          lastModified: media.lastModified || Date.now()
        });
        
        // Create object URL and extract metadata
        const url = URL.createObjectURL(file);
        
        return new Promise((resolve) => {
          const video = document.createElement('video');
          video.preload = 'metadata';
          video.src = url;
          
          video.onloadedmetadata = () => {
            resolve({
              id: media.id,
              type: media.type || 'video',
              filename: media.filename,
              path: filePath,
              size: file.size,
              duration: video.duration,
              width: video.videoWidth,
              height: video.videoHeight,
              lastModified: file.lastModified,
              file: file,
              url: url,
              restored: false
            });
          };
          
          video.onerror = () => {
            resolve({
              ...media,
              file: file,
              url: url,
              restored: false,
              restoreError: 'Failed to load video metadata'
            });
          };
          
          video.load();
        });
      } catch (e) {
        console.error(`❌ Error restoring file ${media.filename}:`, e);
        return {
          ...media,
          file: null,
          url: null,
          restored: true,
          restoreError: e.message
        };
      }
    }));
    
    setImportedMedia(restoredMedia);
  } catch (e) {
    console.error('❌ Desktop file restoration error:', e);
  }
};
```

## 🔧 **TECHNICAL IMPLEMENTATION DETAILS**

### **Architecture Changes**

```
Electron App Structure:
├── electron/
│   ├── main.js          # Main process (Node.js)
│   └── preload.js       # Preload script (IPC bridge)
├── Xinema/
│   ├── frontend/        # React app (Renderer process)
│   │   └── build/       # Production build
│   └── backend/         # Express server (Node.js child process)
└── assets/              # App icons and resources
```

### **IPC Communication Flow**

```
Frontend (Renderer) ←→ Preload (IPC Bridge) ←→ Main Process (Electron APIs)
```

**Example:**
1. Frontend calls `window.electronAPI.readFile(path)`
2. Preload forwards to main via `ipcRenderer.invoke()`
3. Main process reads file using Node.js `fs`
4. Main process returns data via IPC
5. Preload returns to frontend

### **File System Access Patterns**

**Browser (Current):**
- Uses File System Access API (`showOpenFilePicker`, `showSaveFilePicker`)
- Requires user gesture
- Limited to selected files

**Electron (New):**
- Uses native dialogs (`dialog.showOpenDialog`, `dialog.showSaveDialog`)
- Direct file system access via Node.js `fs`
- No user gesture restrictions
- Full file system access
